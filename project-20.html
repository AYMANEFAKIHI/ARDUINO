Project #20: Capacitive Touch Sensor (TTP223)

Add touch input to your projects! The TTP223 module lets you control things with a simple touch - no mechanical button needed. Same technology as your smartphone screen!

4. The Code (Arduino Sketch)

Copy and paste the code below into your Arduino IDE.

/*
 * Project #20: Capacitive Touch Sensor (TTP223)
 * The Arduino Student Lab
 * 
 * Touch the sensor to toggle the LED on/off
 * Demonstrates capacitive touch input
 */

// ============================================
// PIN DEFINITIONS
// ============================================
const int TOUCH_PIN = 2;    // Touch sensor output
const int LED_PIN = 13;     // LED to control

// ============================================
// STATE VARIABLES
// ============================================
bool ledState = false;           // Current LED state
bool lastTouchState = LOW;       // Previous touch reading
unsigned long lastTouchTime = 0; // For debouncing
const unsigned long DEBOUNCE_DELAY = 200;  // 200ms debounce

// Touch counter for fun statistics
int touchCount = 0;

// ============================================
// SETUP - Runs once at startup
// ============================================
void setup() {
  pinMode(TOUCH_PIN, INPUT);
  pinMode(LED_PIN, OUTPUT);
  
  digitalWrite(LED_PIN, LOW);
  
  Serial.begin(9600);
  Serial.println("=== CAPACITIVE TOUCH SENSOR ===");
  Serial.println("Touch the sensor to toggle the LED!");
  Serial.println("");
}

// ============================================
// MAIN LOOP - Runs repeatedly
// ============================================
void loop() {
  // Read touch sensor
  bool touchState = digitalRead(TOUCH_PIN);
  unsigned long currentTime = millis();
  
  // Detect touch (rising edge with debounce)
  if (touchState == HIGH && lastTouchState == LOW) {
    if (currentTime - lastTouchTime > DEBOUNCE_DELAY) {
      lastTouchTime = currentTime;
      touchCount++;
      
      // Toggle LED
      ledState = !ledState;
      digitalWrite(LED_PIN, ledState);
      
      // Report
      Serial.print("Touch #");
      Serial.print(touchCount);
      Serial.print(" - LED is now: ");
      Serial.println(ledState ? "ON" : "OFF");
    }
  }
  
  lastTouchState = touchState;
}

/*
 * HOW CAPACITIVE TOUCH WORKS
 * 
 * Your body is like a capacitor (stores electrical charge).
 * When you touch a conductive surface:
 * 
 *   Before Touch:          After Touch:
 *   +------------+         +------------+
 *   |   Sensor   |         |   Sensor   |
 *   |  Electrode |         |  Electrode |
 *   +------------+         +------||----+
 *         |                       ||
 *     No change              Your finger
 *                           adds capacitance!
 * 
 * The TTP223 chip detects this capacitance change
 * and outputs HIGH when touched.
 * 
 * TTP223 MODULE FEATURES:
 * - Operating voltage: 2V - 5.5V
 * - Output: Digital HIGH (touched) / LOW (not touched)
 * - Response time: ~60ms
 * - Auto-calibration on power-up
 * 
 * MODES (set by solder jumpers on back):
 * - A jumper: Toggle mode (touch = flip output)
 * - B jumper: Momentary mode (HIGH only while touching)
 * - Default: Momentary (no jumpers)
 * 
 * WHY USE CAPACITIVE TOUCH?
 * - No moving parts (no wear)
 * - Can work through thin materials
 * - Sleek, modern interface
 * - Water resistant options available
 */
