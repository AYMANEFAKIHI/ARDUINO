Project #28: 4-Digit Clock Display (TM1637)

Build a digital clock using the popular TM1637 4-digit display module! This simple-to-wire display is perfect for clocks, counters, and score displays.

  • Building a simple timekeeping system

4. The Code (Arduino Sketch)

Copy and paste the code below into your Arduino IDE.

/*
 * Project #28: 4-Digit Clock Display (TM1637)
 * The Arduino Student Lab
 * 
 * Simple clock using TM1637 display module
 * Shows hours:minutes with blinking colon
 */

#include <TM1637Display.h>

// ============================================
// PIN DEFINITIONS
// ============================================
const int CLK_PIN = 2;
const int DIO_PIN = 3;

// ============================================
// CREATE DISPLAY OBJECT
// ============================================
TM1637Display display(CLK_PIN, DIO_PIN);

// ============================================
// TIME VARIABLES
// ============================================
int hours = 12;
int minutes = 0;
int seconds = 0;
unsigned long lastSecond = 0;
bool colonOn = true;

// ============================================
// SETUP
// ============================================
void setup() {
  // Set brightness (0-7)
  display.setBrightness(5);
  
  Serial.begin(9600);
  Serial.println("=== TM1637 CLOCK DISPLAY ===");
  Serial.println("Time starts at 12:00");
  Serial.println("Use Serial Monitor to set time:");
  Serial.println("Send 'H+' or 'H-' for hours");
  Serial.println("Send 'M+' or 'M-' for minutes");
  Serial.println("");
  
  updateDisplay();
}

// ============================================
// MAIN LOOP
// ============================================
void loop() {
  unsigned long currentMillis = millis();
  
  // Update every second
  if (currentMillis - lastSecond >= 1000) {
    lastSecond = currentMillis;
    
    // Increment time
    seconds++;
    if (seconds >= 60) {
      seconds = 0;
      minutes++;
      
      if (minutes >= 60) {
        minutes = 0;
        hours++;
        
        if (hours >= 24) {
          hours = 0;
        }
      }
      
      updateDisplay();
    }
    
    // Toggle colon every second
    colonOn = !colonOn;
    updateDisplay();
    
    // Print time to serial
    printTime();
  }
  
  // Handle serial commands for time setting
  if (Serial.available()) {
    String cmd = Serial.readStringUntil('\n');
    cmd.trim();
    
    if (cmd == "H+") {
      hours = (hours + 1) % 24;
      updateDisplay();
    }
    else if (cmd == "H-") {
      hours = (hours + 23) % 24;
      updateDisplay();
    }
    else if (cmd == "M+") {
      minutes = (minutes + 1) % 60;
      seconds = 0;
      updateDisplay();
    }
    else if (cmd == "M-") {
      minutes = (minutes + 59) % 60;
      seconds = 0;
      updateDisplay();
    }
  }
}

// ============================================
// HELPER FUNCTIONS
// ============================================

void updateDisplay() {
  // Create the display value: HHMM
  int displayValue = hours * 100 + minutes;
  
  // Show on display with or without colon
  uint8_t colonMask = colonOn ? 0b01000000 : 0;
  
  display.showNumberDecEx(displayValue, colonMask, true);
}

void printTime() {
  Serial.print(hours < 10 ? "0" : "");
  Serial.print(hours);
  Serial.print(":");
  Serial.print(minutes < 10 ? "0" : "");
  Serial.print(minutes);
  Serial.print(":");
  Serial.print(seconds < 10 ? "0" : "");
  Serial.println(seconds);
}

/*
 * TM1637 DISPLAY OVERVIEW
 * 
 * The TM1637 is a dedicated display driver IC
 * that handles all the segment multiplexing.
 * 
 * Features:
 * - 4 digits with optional colon (:)
 * - Only 2 data pins needed (CLK, DIO)
 * - Adjustable brightness (8 levels)
 * - Built-in key scanning (some modules)
 * 
 * MODULE LAYOUT:
 *   +--+--+--+--+
 *   |  |: |  |  |
 *   +--+--+--+--+
 * 
 * LIBRARY FUNCTIONS:
 * 
 * display.setBrightness(0-7)
 *   Set display brightness
 * 
 * display.showNumberDec(num)
 *   Show a number (0-9999)
 * 
 * display.showNumberDecEx(num, dots, leadingZeros)
 *   Show number with colon control
 *   dots: 0b01000000 for center colon
 * 
 * display.setSegments(segments[])
 *   Show custom segment patterns
 * 
 * display.clear()
 *   Turn off all segments
 * 
 * NOTE: This is a SIMPLE clock - time drifts!
 * For accurate timekeeping, use RTC module (DS3231)
 * See Project #52 for RTC version
 */

  1. Two-Wire Protocol: TM1637 uses CLK and DIO for communication.
  2. Library Abstraction: TM1637Display library handles protocol.
  3. Colon Control: Bit mask toggles center colon on/off.
  4. Simple Timekeeping: millis() counts seconds (drifts over time).
