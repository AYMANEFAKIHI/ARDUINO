Project #23: Parking Assist (3-Zone LED)

Build a car parking sensor! Three colored LEDs indicate distance zones - green (safe), yellow (caution), red (stop!). Add a buzzer for that authentic parking sensor experience.

1. Concept and Description

In this project, we will explore the following concepts:

  • Using distance thresholds for zones
  • Multi-level visual feedback
  • Variable-rate audio alerts
  • Practical automotive applications

4. The Code (Arduino Sketch)

Copy and paste the code below into your Arduino IDE.

/*
 * Project #23: Parking Assist (3-Zone LED)
 * The Arduino Student Lab
 * 
 * Visual and audio feedback based on distance
 * Green = Safe, Yellow = Caution, Red = Stop!
 */

// ============================================
// PIN DEFINITIONS
// ============================================
const int TRIG_PIN = 9;
const int ECHO_PIN = 10;
const int GREEN_LED = 2;
const int YELLOW_LED = 3;
const int RED_LED = 4;
const int BUZZER_PIN = 8;

// ============================================
// DISTANCE ZONES (in centimeters)
// ============================================
const int ZONE_FAR = 100;     // Beyond this = all clear
const int ZONE_CAUTION = 50;  // Yellow zone starts
const int ZONE_DANGER = 20;   // Red zone starts
const int ZONE_STOP = 10;     // Too close!

// ============================================
// BUZZER TIMING (beep intervals in ms)
// ============================================
const int BEEP_FAR = 1000;      // Slow beep when far
const int BEEP_CAUTION = 500;   // Medium beep
const int BEEP_DANGER = 200;    // Fast beep
const int BEEP_STOP = 0;        // Continuous

// ============================================
// STATE VARIABLES
// ============================================
unsigned long lastBeepTime = 0;
bool buzzerOn = false;

// ============================================
// SETUP
// ============================================
void setup() {
  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);
  pinMode(GREEN_LED, OUTPUT);
  pinMode(YELLOW_LED, OUTPUT);
  pinMode(RED_LED, OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);
  
  Serial.begin(9600);
  Serial.println("=== PARKING ASSIST SYSTEM ===");
  Serial.println("Zones: >100cm=Clear, 50-100=Caution, 20-50=Danger, <20=STOP!");
  Serial.println("");
}

// ============================================
// MAIN LOOP
// ============================================
void loop() {
  // Measure distance
  float distance = measureDistance();
  
  // Determine zone and update display
  if (distance > ZONE_FAR) {
    // All clear - far away
    setLEDs(LOW, LOW, LOW);
    noTone(BUZZER_PIN);
    Serial.print("CLEAR: ");
  }
  else if (distance > ZONE_CAUTION) {
    // Green zone - approaching
    setLEDs(HIGH, LOW, LOW);
    handleBeeper(BEEP_FAR);
    Serial.print("GREEN: ");
  }
  else if (distance > ZONE_DANGER) {
    // Yellow zone - caution
    setLEDs(HIGH, HIGH, LOW);
    handleBeeper(BEEP_CAUTION);
    Serial.print("YELLOW: ");
  }
  else if (distance > ZONE_STOP) {
    // Red zone - danger
    setLEDs(HIGH, HIGH, HIGH);
    handleBeeper(BEEP_DANGER);
    Serial.print("RED: ");
  }
  else {
    // Too close - STOP!
    setLEDs(HIGH, HIGH, HIGH);  // All on
    tone(BUZZER_PIN, 2000);     // Continuous alarm
    Serial.print("STOP!: ");
  }
  
  Serial.print(distance, 1);
  Serial.println(" cm");
  
  delay(100);
}

// ============================================
// HELPER FUNCTIONS
// ============================================

float measureDistance() {
  // Send trigger pulse
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);
  
  // Measure echo pulse
  long duration = pulseIn(ECHO_PIN, HIGH, 30000);
  
  // Convert to distance
  float distance = duration * 0.0343 / 2.0;
  
  // Handle timeout
  if (distance == 0 || distance > 400) {
    distance = 999;  // Out of range
  }
  
  return distance;
}

void setLEDs(bool green, bool yellow, bool red) {
  digitalWrite(GREEN_LED, green);
  digitalWrite(YELLOW_LED, yellow);
  digitalWrite(RED_LED, red);
}

void handleBeeper(int interval) {
  unsigned long currentTime = millis();
  
  if (currentTime - lastBeepTime >= interval) {
    lastBeepTime = currentTime;
    
    if (buzzerOn) {
      noTone(BUZZER_PIN);
      buzzerOn = false;
    } else {
      tone(BUZZER_PIN, 1500, interval / 2);
      buzzerOn = true;
    }
  }
}

/*
 * PARKING SENSOR DESIGN NOTES
 * 
 * Zone Design:
 *   CLEAR    |  GREEN  | YELLOW |  RED  | STOP!
 *   >100cm   | 50-100  | 20-50  | 10-20 | <10cm
 *            |    ‚óè    |   ‚óè‚óè   |  ‚óè‚óè‚óè  | ‚óè‚óè‚óè + ALARM

5. Explanation & Challenges

üîç How It Works

  1. Distance Zones: Distance thresholds define 4 zones (clear, green, yellow, red).
  2. LED Stacking: More LEDs light as you get closer.
  3. Variable Beeping: Beep interval decreases with distance.
  4. Continuous Alarm: Solid tone when too close.

  • Dual Sensors: Left and right sensors for corner detection.
  • NeoPixel Bar: Smooth color gradient from green to red.
  • Data Logging: Record minimum distances during parking sessions.
