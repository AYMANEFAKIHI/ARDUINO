Project #85: Blynk Sensor Data Monitoring

View sensor readings (temperature, humidity) on your phone via Blynk.

1. Concept and Description

Send data from the ESP8266 to the Blynk cloud and display it on a gauge or graph widget.

2. Components Needed

  • ESP8266
  • DHT11 Sensor

3. Physical Montage (Wiring Instructions)

DHT11 to D2 (GPIO 4).

4. The Code (Arduino Sketch)

Copy and paste the code below into your Arduino IDE.

#include 
#include 

#define DHTPIN 4
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);
BlynkTimer timer;

void sendSensor() {
  float h = dht.readHumidity();
  float t = dht.readTemperature();
  Blynk.virtualWrite(V5, h);
  Blynk.virtualWrite(V6, t);
}

void setup() {
  Blynk.begin("AUTH", "SSID", "PASS");
  dht.begin();
  timer.setInterval(1000L, sendSensor);
}

void loop() {
  Blynk.run();
  timer.run();
}
