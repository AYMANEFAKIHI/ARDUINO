Project #52: Real-Time Clock (DS3231) Display

Keep accurate time even when power is off using a DS3231 RTC module.

1. Concept and Description

The DS3231 is a highly accurate Real-Time Clock (RTC) with an integrated temperature-compensated crystal oscillator (TCXO) and crystal.

2. Components Needed

  • Arduino Uno
  • DS3231 RTC Module
  • Jumper Wires

3. Physical Montage (Wiring Instructions)

VCC to 5V, GND to GND, SDA to A4, SCL to A5.

4. The Code (Arduino Sketch)

Copy and paste the code below into your Arduino IDE.


#include 
#include "RTClib.h"

RTC_DS3231 rtc;

void setup() {
  Serial.begin(9600);
  if (!rtc.begin()) {
    Serial.println("Couldn't find RTC");
    while (1);
  }
  
  if (rtc.lostPower()) {
    Serial.println("RTC lost power, lets set the time!");
    rtc.adjust(DateTime(F(__DATE__), F(__TIME__)));
  }
}

void loop() {
  DateTime now = rtc.now();
  
  Serial.print(now.year(), DEC);
  Serial.print('/');
  Serial.print(now.month(), DEC);
  Serial.print('/');
  Serial.print(now.day(), DEC);
  Serial.print(" ");
  Serial.print(now.hour(), DEC);
  Serial.print(':');
  Serial.print(now.minute(), DEC);
  Serial.print(':');
  Serial.print(now.second(), DEC);
  Serial.println();
  
  delay(1000);
}
