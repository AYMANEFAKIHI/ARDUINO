Project #60: Rotary Encoder Basic Counter

Use a rotary encoder to count up and down by turning a knob.

1. Concept and Description

Rotary encoders are different from potentiometers; they can spin infinitely and provide digital pulses to indicate direction and steps.

2. Components Needed

  • Arduino Uno
  • Rotary Encoder Module

3. Physical Montage (Wiring Instructions)

CLK to D2, DT to D3, SW to D4, + to 5V, GND to GND.

4. The Code (Arduino Sketch)

Copy and paste the code below into your Arduino IDE.

#define CLK 2
#define DT 3
#define SW 4

int counter = 0;
int currentStateCLK;
int lastStateCLK;

void setup() {
  pinMode(CLK,INPUT);
  pinMode(DT,INPUT);
  pinMode(SW, INPUT_PULLUP);
  Serial.begin(9600);
  lastStateCLK = digitalRead(CLK);
}

void loop() {
  currentStateCLK = digitalRead(CLK);

  if (currentStateCLK != lastStateCLK  && currentStateCLK == 1){
    if (digitalRead(DT) != currentStateCLK) {
      counter --;
    } else {
      counter ++;
    }
    Serial.print("Counter: ");
    Serial.println(counter);
  }

  lastStateCLK = currentStateCLK;
  
  if(digitalRead(SW) == LOW) {
    counter = 0;
    Serial.println("Reset");
    delay(200);
  }
}
