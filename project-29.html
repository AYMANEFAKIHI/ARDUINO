Project #29: LCD Hello World (16x2)

Master the classic 16x2 LCD display! Learn to display text, scroll messages, and create custom characters. The LCD is essential for projects that need to show information.

1. Concept and Description

In this project, we will explore the following concepts:

  • Wiring a parallel LCD display
  • Using the LiquidCrystal library
  • LCD cursor positioning
  • Creating custom characters

4. The Code (Arduino Sketch)

Copy and paste the code below into your Arduino IDE.

/*
 * Project #29: LCD Hello World (16x2)
 * The Arduino Student Lab
 * 
 * Displays text on a 16x2 LCD display
 * Demonstrates positioning, scrolling, and custom chars
 */

#include <LiquidCrystal.h>

// ============================================
// LCD PIN CONFIGURATION
// ============================================
// LiquidCrystal(RS, E, D4, D5, D6, D7)
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

// ============================================
// CUSTOM CHARACTER DEFINITIONS
// ============================================
// Each character is 5x8 pixels (5 columns, 8 rows)
byte smiley[8] = {
  0b00000,
  0b01010,
  0b01010,
  0b00000,
  0b10001,
  0b01110,
  0b00000,
  0b00000
};

byte heart[8] = {
  0b00000,
  0b01010,
  0b11111,
  0b11111,
  0b11111,
  0b01110,
  0b00100,
  0b00000
};

// ============================================
// SETUP
// ============================================
void setup() {
  // Initialize LCD (16 columns, 2 rows)
  lcd.begin(16, 2);
  
  // Create custom characters (indices 0-7)
  lcd.createChar(0, smiley);
  lcd.createChar(1, heart);
  
  Serial.begin(9600);
  Serial.println("=== LCD HELLO WORLD ===");
  
  // Show startup message
  lcd.print("Hello, Arduino!");
  lcd.setCursor(0, 1);
  lcd.print("LCD Display ");
  lcd.write(byte(0));  // Smiley
  
  delay(3000);
}

// ============================================
// MAIN LOOP
// ============================================
void loop() {
  // Demo 1: Basic text
  lcd.clear();
  lcd.print("Demo 1: Text");
  lcd.setCursor(0, 1);
  lcd.print("16 chars x 2 row");
  delay(2000);
  
  // Demo 2: Cursor positioning
  lcd.clear();
  lcd.print("Demo 2: Position");
  delay(1000);
  
  lcd.setCursor(0, 1);  // Column 0, Row 1
  for (int col = 0; col < 16; col++) {
    lcd.setCursor(col, 1);
    lcd.print("*");
    delay(100);
  }
  delay(1000);
  
  // Demo 3: Scrolling
  lcd.clear();
  lcd.print("Demo 3: Scroll");
  lcd.setCursor(0, 1);
  lcd.print("Watch this!");
  delay(1000);
  
  // Scroll left
  for (int i = 0; i < 5; i++) {
    lcd.scrollDisplayLeft();
    delay(300);
  }
  // Scroll right
  for (int i = 0; i < 5; i++) {
    lcd.scrollDisplayRight();
    delay(300);
  }
  delay(1000);
  
  // Demo 4: Custom characters
  lcd.clear();
  lcd.print("Demo 4: Custom");
  lcd.setCursor(0, 1);
  lcd.print("I ");
  lcd.write(byte(1));  // Heart
  lcd.print(" Arduino ");
  lcd.write(byte(0));  // Smiley
  delay(3000);
  
  // Demo 5: Counter
  lcd.clear();
  lcd.print("Demo 5: Counter");
  for (int i = 0; i <= 100; i += 10) {
    lcd.setCursor(0, 1);
    lcd.print("Value: ");
    lcd.print(i);
    lcd.print("%  ");  // Extra spaces clear old chars
    delay(300);
  }
  delay(2000);
}

/*
 * LCD LIBRARY REFERENCE
 * 
 * INITIALIZATION:
 * lcd.begin(cols, rows)  - Initialize display size
 * 
 * TEXT OUTPUT:
 * lcd.print("text")      - Print text at cursor
 * lcd.print(number)      - Print number
 * lcd.write(byte)        - Print single character
 * 
 * CURSOR CONTROL:
 * lcd.setCursor(col,row) - Move cursor (0-indexed)
 * lcd.home()             - Cursor to top-left
 * lcd.cursor()           - Show cursor
 * lcd.noCursor()         - Hide cursor
 * lcd.blink()            - Blinking cursor
 * lcd.noBlink()          - Stop blinking
 * 
 * DISPLAY CONTROL:
 * lcd.clear()            - Clear and cursor home
 * lcd.display()          - Turn on display
 * lcd.noDisplay()        - Turn off display
 * lcd.scrollDisplayLeft()
 * lcd.scrollDisplayRight()
 * 
 * CUSTOM CHARACTERS:
 * lcd.createChar(num, array)  - Define custom (0-7)
 * lcd.write(byte(num))        - Display custom char
 * 
 * CURSOR POSITIONS:
 * +--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+
 * |00|01|02|03|04|05|06|07|08|09|10|11|12|13|14|15| Row 0
 * +--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+
 * |00|01|02|03|04|05|06|07|08|09|10|11|12|13|14|15| Row 1
 * +--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+
 */

  4. Scrolling: Built-in functions shift entire display content.
