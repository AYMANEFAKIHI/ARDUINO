Project #82: Getting Weather Data (API)

Fetch live weather data from OpenWeatherMap API.

1. Concept and Description

Learn how to make HTTP GET requests to an API and parse the JSON response.

2. Components Needed

  • ESP8266

3. Physical Montage (Wiring Instructions)

None.

4. The Code (Arduino Sketch)

Copy and paste the code below into your Arduino IDE.

#include 
#include 

void setup() {
  Serial.begin(115200);
  WiFi.begin("SSID", "PASS");
}

void loop() {
  if (WiFi.status() == WL_CONNECTED) {
    WiFiClient client;
    HTTPClient http;
    http.begin(client, "http://api.openweathermap.org/data/2.5/weather?q=London&appid=YOUR_API_KEY");
    int httpCode = http.GET();
    if (httpCode > 0) {
      String payload = http.getString();
      Serial.println(payload);
    }
    http.end();
  }
  delay(60000);
}
