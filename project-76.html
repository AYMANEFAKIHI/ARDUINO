Project #76: RFID Reader (RC522)

Read unique IDs from RFID tags and cards.

1. Concept and Description

RFID (Radio Frequency Identification) is used for access control, inventory, and payments.

2. Components Needed

  • Arduino Uno
  • RC522 RFID Module
  • Tags/Cards

3. Physical Montage (Wiring Instructions)

SDA(SS) to D10, SCK to D13, MOSI to D11, MISO to D12, RST to D9. 3.3V Power!

4. The Code (Arduino Sketch)

Copy and paste the code below into your Arduino IDE.


#include 
#include 

#define SS_PIN 10
#define RST_PIN 9
MFRC522 mfrc522(SS_PIN, RST_PIN);

void setup() {
  Serial.begin(9600);
  SPI.begin();
  mfrc522.PCD_Init();
}

void loop() {
  if ( ! mfrc522.PICC_IsNewCardPresent()) return;
  if ( ! mfrc522.PICC_ReadCardSerial()) return;

  Serial.print("UID tag :");
  String content= "";
  for (byte i = 0; i < mfrc522.uid.size; i++) {
     Serial.print(mfrc522.uid.uidByte[i] < 0x10 ? " 0" : " ");
     Serial.print(mfrc522.uid.uidByte[i], HEX);
  }
  Serial.println();
  delay(1000);
}
