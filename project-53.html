Project #53: LCD Stopwatch with Buttons

Create a stopwatch with Start, Stop, and Reset functionality displayed on an LCD.

1. Concept and Description

This project combines button handling, time calculation, and LCD display updates to create a functional stopwatch.

2. Components Needed

  • Arduino Uno
  • 16x2 LCD
  • 2 Push Buttons
  • Resistors (10k)
  • Jumper Wires

3. Physical Montage (Wiring Instructions)

LCD as standard. Start/Stop Button to D2, Reset Button to D3.

4. The Code (Arduino Sketch)

Copy and paste the code below into your Arduino IDE.


#include 

LiquidCrystal lcd(7, 8, 9, 10, 11, 12);
const int startStopPin = 2;
const int resetPin = 3;

unsigned long startTime;
unsigned long elapsedTime = 0;
boolean running = false;

void setup() {
  lcd.begin(16, 2);
  pinMode(startStopPin, INPUT_PULLUP);
  pinMode(resetPin, INPUT_PULLUP);
  lcd.print("Stopwatch");
}

void loop() {
  if (digitalRead(startStopPin) == LOW) {
    delay(200); // Debounce
    if (running) {
      elapsedTime += millis() - startTime;
      running = false;
    } else {
      startTime = millis();
      running = true;
    }
  }
  
  if (digitalRead(resetPin) == LOW) {
    running = false;
    elapsedTime = 0;
    lcd.clear();
    lcd.print("Stopwatch");
  }
  
  if (running) {
    unsigned long currentMillis = millis();
    unsigned long totalTime = elapsedTime + (currentMillis - startTime);
    
    lcd.setCursor(0, 1);
    lcd.print(totalTime / 1000.0);
  }
}
